    ```matlab
    function test_dashboard_widget_button_bar()
    %TEST_DASHBOARD_WIDGET_BUTTON_BAR Regression: bar is right-anchored small strip.
    %
    % Asserts the per-widget button bar (Tag='WidgetButtonBar') is rendered
    % as a small right-anchored strip (<= 64px wide) and not as a full-width
    % header band that would obscure each widget's own title rendering.

        add_test_paths();

        fig = figure('Visible', 'off');
        cleanup = onCleanup(@() close(fig));

        d = DashboardEngine('Parent', fig);
        d.addWidget(TextWidget('Key', 'tw1', 'Title', 'Hello', 'Body', 'World'), ...
            [1 1 6 4]);
        d.render();

        % Find the WidgetButtonBar uipanel.
        bar = findobj(fig, 'Tag', 'WidgetButtonBar');
        assert(~isempty(bar), 'WidgetButtonBar not found');
        bar = bar(1);

        oldUnits = get(bar, 'Units');
        set(bar, 'Units', 'pixels');
        barPos = get(bar, 'Position');
        set(bar, 'Units', oldUnits);

        panelHandle = get(bar, 'Parent');
        oldPU = get(panelHandle, 'Units');
        set(panelHandle, 'Units', 'pixels');
        pp = get(panelHandle, 'Position');
        set(panelHandle, 'Units', oldPU);

        inset = 2;
        % 1. Width is small, not full panel width.
        assert(barPos(3) <= 64, ...
            sprintf('bar width %g exceeds 64px', barPos(3)));
        % 2. x anchor is NOT at left edge (inset = 2). Allow tolerance for
        %    very narrow panels where bar collapses; require >2 in normal case.
        if pp(3) > 64 + 2 * inset
            assert(barPos(1) > inset, ...
                sprintf('bar not right-anchored: x=%g (expected > %g)', barPos(1), inset));
        end
        % 3. Right edge near panel right edge (within 1px of inset).
        rightGap = pp(3) - (barPos(1) + barPos(3));
        assert(abs(rightGap - inset) <= 1, ...
            sprintf('bar right-edge gap %g not ~%g', rightGap, inset));

        % 4. Buttons fit inside the bar.
        det  = findobj(bar, 'Tag', 'DetachButton',   '-depth', 1);
        info = findobj(bar, 'Tag', 'InfoIconButton', '-depth', 1);
        if ~isempty(det)
            dp = get(det(1), 'Position');
            assert(dp(1) + dp(3) <= barPos(3) + 0.5, ...
                'DetachButton overflows bar width');
        end
        if ~isempty(info)
            ip = get(info(1), 'Position');
            assert(ip(1) + ip(3) <= barPos(3) + 0.5, ...
                'InfoIconButton overflows bar width');
        end

        fprintf('    All button-bar tests passed.\n');
    end

    function add_test_paths()
        thisDir = fileparts(mfilename('fullpath'));
        repoRoot = fullfile(thisDir, '..');
        run(fullfile(repoRoot, 'install.m'));
    end
    ```
